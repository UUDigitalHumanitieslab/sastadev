def astadefaultfilter(query: Query, xrs: ExactResultsDict, xr: ExactResult) -> bool:
    return query.process == pre_process or \
    (implies('A029' in xrs, xr not in xrs['A029'])
     and implies('A045' in xrs, xr not in xrs['A045']))
